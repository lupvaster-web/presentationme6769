def prost(x):
    return x>1 and all(x %i !=0 for i in range(2, int(x**0.5)+1))


def de(x):
    d= set()
    for i in range(2, int(x**0.5)+1):
        if x%i ==0:
            d.add(i)
            d.add(x//i)
    return sorted(d)


for x in range(174457, 174506):
    p= de(x)
    if len(p)==2:
        print(p[0],p[1])
