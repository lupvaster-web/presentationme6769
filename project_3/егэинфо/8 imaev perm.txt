from itertools import*
k=0
for x in product('012345678', repeat=5):
    s= ''.join(x)
    if s[0] not in'01357' and s[-1]not in '18' and s.count('3')==1:
        s=s.replace('2','*').replace('3','*').replace('4','*').replace('5','*')
        if '*6'not in s and '6*'not in s:
            k+=1
print(k)
